Quick Reference (Preview)

selection_preview.py 
# Simple if statement
age = 15
if age >= 13:
    print("You're a teenager!")

# if-else: two options
score = 45
if score >= 50:
    print("Pass")
else:
    print("Fail")

# if-elif-else: multiple options
grade = 75
if grade >= 90:
    print("A")
elif grade >= 70:
    print("B")
elif grade >= 50:
    print("C")
else:
    print("Fail")
